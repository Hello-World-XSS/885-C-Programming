```c
#include <math.h>
int main(){
    double e = 2,y;
    long n = 1;
    while (e-y>1e-6){
        y = e;
        ++n;
```

- 方法二：
```c
#include <stdio.h>
#include <math.h>
int main() {
   double a = 2,y;
   long int n=1;
   do{
       y=a;
       n+=1;
       a=pow(1+1/(double)n,(double)n);
   }while(a-y>1e-6);
   printf("e = %f\nn = %ld\n",a,n);
   return 0;
}
```
